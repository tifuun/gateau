25/**

36__device__ __inline__ void scanPoint(AzEl* center, AzEl* out, bool chop, float sep = 0.) {
37 float offset = 0.;

39 if (chop) {
40 offset = sep;

43 out->Az = center->Az + offset;
44 out->El = center->El;
45}
